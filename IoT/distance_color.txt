void readColor() {
  // RED
  digitalWrite(S2, LOW);
  digitalWrite(S3, LOW);
  redFrequency = pulseIn(OUT, LOW);
  redFrequency = map(redFrequency, 10, 60, 255, 0);

  // GREEN
  digitalWrite(S2, HIGH);
  digitalWrite(S3, HIGH);
  greenFrequency = pulseIn(OUT, LOW);
  greenFrequency = map(greenFrequency, 10, 60, 255, 0);

  // BLUE
  digitalWrite(S2, LOW);
  digitalWrite(S3, HIGH);
  blueFrequency = pulseIn(OUT, LOW);
  blueFrequency = map(blueFrequency, 10, 60, 255, 0);

  Serial.print("R: "); Serial.print(redFrequency);
  Serial.print(" G: "); Serial.print(greenFrequency);
  Serial.print(" B: "); Serial.println(blueFrequency);
}

// Function to detect and print color name
void detectColor() {
  if (redFrequency > greenFrequency && redFrequency > blueFrequency) {
    Serial.println("Detected Color: RED");
  } 
  else if (greenFrequency > redFrequency && greenFrequency > blueFrequency) {
    Serial.println("Detected Color: GREEN");
  } 
  else if (blueFrequency > redFrequency && blueFrequency > greenFrequency) {
    Serial.println("Detected Color: BLUE");
  } 
  else {
    Serial.println("Detected Color: UNKNOWN/WHITE");
  }
}
